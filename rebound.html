Rebound to model astrophysics

Why I decided to use Rebound

As I started to research more into astrophysics I started to realise the importance of computer programming in order to handle and sort through data as well as presenting it in a way that could return understandable information, because of this I downloaded rebound and started playing around with values coming up with a simulator for 0.3 to 1 mass ratio of a star and planet to map the planets elliptical orbit shown below.

Simulation Code

# REBOUND N-Body Simulation for Planetary Orbits
# Simulates elliptical orbits with varying mass ratios

import rebound
import numpy as np
import matplotlib.pyplot as plt

def setup_simulation(star_mass=1.0, planet_mass=0.001):
# Create a new simulation
sim = rebound.Simulation()
sim.integrator = "ias15" # High accuracy integrator
sim.G = 1.0 # Gravitational constant in appropriate units

# Add central star
sim.add(m=star_mass)

# Add planet with elliptical orbit
sim.add(m=planet_mass, a=1.0, e=0.1)

# Move to center of mass frame
sim.move_to_com()

return sim

def run_simulation(sim, duration=100):
# Set up arrays to store results
times = np.linspace(0, duration*2*np.pi, 1000)
positions = []
energies = []

# Initial energy
E0 = sim.calculate_energy()

for i, t in enumerate(times):
sim.integrate(t)

# Store planet position
planet = sim.particles[1]
positions.append([planet.x, planet.y])

# Calculate and store energy conservation
E = sim.calculate_energy()
energies.append(abs((E - E0)/E0))

return np.array(positions), np.array(energies), times

def plot_results(positions, energies, times):
# Create figure with subplots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))

# Plot orbital trajectory
ax1.plot(positions[:, 0], positions[:, 1], 'b-', linewidth=1)
ax1.plot(0, 0, 'yo', markersize=10, label='Star')
ax1.set_xlabel('X Position')
ax1.set_ylabel('Y Position')
ax1.set_title('Elliptical Orbit')
ax1.legend()
ax1.grid(True, alpha=0.3)
ax1.set_aspect('equal')

# Plot energy conservation
ax2.semilogy(times, energies)
ax2.set_xlabel('Time')
ax2.set_ylabel('Relative Energy Error')
ax2.set_title('Energy Conservation Test')
ax2.grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('rebound_results.png', dpi=300, bbox_inches='tight')
plt.show()

# Main execution
if __name__ == "__main__":
# Test different mass ratios from 0.3 to 1
mass_ratios = np.linspace(0.3, 1.0, 5)

for ratio in mass_ratios:
print(f"Running simulation with mass ratio: {ratio:.2f}")
sim = setup_simulation(planet_mass=ratio*0.001)
positions, energies, times = run_simulation(sim)
plot_results(positions, energies, times)
